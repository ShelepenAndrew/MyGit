#include <iostream>
#include <conio.h> // библиотека, необходимая для работы оператора getch();
using namespace std;

int main()
{
	setlocale(0, "");

	int a,b,c;
	cout << "Введите первое число: ";
	cin >> a;

	cout << "Введите второе число ";
	cin >> b;

	cout << endl << "Выберите математическую операцию: " << endl <<
	"1. Сложение" << endl <<
	"2. Вычитание" << endl <<
	"3. Умножение" << endl <<
	"4. Деление" << endl;

	cin >> c;

	switch (c)
	{

	case 1:
		cout << "Результат сложения = "<< a + b << endl;
		break;

	case 2:
		cout << "Результат вычитания = " << a - b << endl;
		break;

	case 3:
		cout << "Результат произведения = " << a * b << endl;
		break;

	case 4:
		cout << "Результат деления = " << (float)a / b << endl;
		break;

	default:
		cout << "Ошибка! Пункт меню отсутствует!" << endl;
		break;
	}

	_getch(); // Команда задержки экрана
	return 0;
}
