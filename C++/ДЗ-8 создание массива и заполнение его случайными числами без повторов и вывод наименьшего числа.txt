#include <iostream>
#include <conio.h> // библиотека, необходимая для работы оператора getch();
#include <ctime> // библиотека, необходимая для работы 	srand(time(NULL));

using namespace std;

int main()
{
	setlocale(0, "");
	srand(time(NULL));
	
	const int SIZE = 10;
	int arr[SIZE];


	bool alreadyThere;
	for (int i = 0; i < SIZE; )
	{
		alreadyThere = false;
		int newRandomValue = rand() % 20;
		for (int j = 0; j < i; j++)
		{
			if (arr[j] == newRandomValue)
			{
				alreadyThere = true;
				break;
			}

		}
	
		if (!alreadyThere)
		{
			arr[i] = newRandomValue;
			i++;
		}
	
	}

	int minValue= arr[0];
	for (int i = 1; i < SIZE; i++)
	{
		if (arr[i]<minValue)
		{
			minValue = arr[i];
		}
	}
	cout << "Наименьшее число в массиве " << minValue << endl;

	_getch(); // Команда задержки экрана
	return 0;
}
