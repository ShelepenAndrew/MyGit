#include <iostream>
#include <ctime> // библиотека, необходима для (time(NULL);
#include <conio.h> // библиотека, необходимая для работы оператора getch();

using namespace std;
int main()
{
	setlocale(LC_ALL, "Rus");

	srand(time(NULL));

	int const SIZE = 10;
	int arr[SIZE];

	for (int i = 0; i < SIZE; i++)
	{
		arr[i] = rand() % 20;
	}

	for (int i = 0; i < SIZE; i++)
	{
		cout << arr[i] << endl;
	}
	
	_getch(); // Команда задержки экрана
	return 0;

}
