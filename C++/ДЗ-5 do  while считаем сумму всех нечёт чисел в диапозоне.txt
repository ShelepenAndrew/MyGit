#include <iostream>
#include <conio.h> // библиотека, необходимая для работы оператора getch();

using namespace std;
int main()
{
	setlocale(LC_ALL, "Rus");

	int sum = 0;
	int rangeBegin, rangeEnd;

	cout << "Введите начало диапазона: ";
	cin >> rangeBegin;
	cout << "Введите конец диапазона: ";
	cin >> rangeEnd;

	do
	{
		if (rangeBegin %2 != 0)
		{
			sum += rangeBegin;
		}
		rangeBegin++;
	} while (rangeBegin<rangeEnd);
		
	cout << "Сумма всех целых нечётных чисел в указанном диапазоне = " << sum << endl;
	_getch(); // Команда задержки экрана
	return 0;
}
