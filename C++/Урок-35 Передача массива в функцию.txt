#include <iostream>
#include <conio.h> // библиотека, необходимая для работы оператора getch();

using namespace std;

void fillArray(int arr[], const int size)
	{
		for (int i = 0; i < size; i++)
		{
			arr[i] = rand() % 10;
		}
	}

void PrintArray(int arr[], const int size)
{
	for (int i = 0; i < size; i++)
	{
		cout << arr[i] << endl;
	}
}

int main()
{
	setlocale(0, "");
	const int size = 10;
	int arr[size];

	fillArray(arr, size);
	PrintArray(arr,size);

	_getch(); // Команда задержки экрана
	return 0;
}
